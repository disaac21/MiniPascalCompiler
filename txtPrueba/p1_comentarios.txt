program IgnorarComentariosSimple;

{ Este programa lee un texto simulando código Mini-Pascal
  e ignora los comentarios y caracteres de espacio en blanco, 
  tabulación y nueva línea, sin usar funciones nativas avanzadas. }

  texto: string;
  i: integer;
  dentroComentario: boolean;
