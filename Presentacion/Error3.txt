program ManejoDeTipos;

{ Este programa maneja variables de tipo integer, char, string, y boolean }

var
  123: integer;      { Variable de tipo integer }
  letra: char;       { Variable de tipo char }
  prueba: integer;
  mensaje: string;   { Variable de tipo string }
  esVerdadero boolean;  { Variable de tipo boolean }

begin
  { Asignaci√≥n de valores a las variables }
  num := 42;
  letra := 'A';
  mensaje := 'Hola Pascal';
  esVerdadero := 'hola';

  { Operaciones con variables de tipo integer }
  num := 10$;  { Sumar 10 al valor de num }

  { Operaciones con variables de tipo boolean }
  if esVerdadero then
    writeln('El valor booleano es verdadero.')
  else
    writeln('El valor booleano es falso.');

  { Mostrar los valores de las variables }
  writeln('Valor de num (integer): ', num);
  writeln('Valor de letra (char): ', letra);
  writeln('Valor de mensaje (string): ', mensaje);
end.
